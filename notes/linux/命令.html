# ffmpeg

去除水印

ffmpeg -i ./1.mp4 -filter_complex "delogo=x=30:y=210:w=500:h=140:" 4.mp4

ffmpeg -i j.flv -c copy j.mp4 将mp4合并

ffmpeg -i ./5.mp4 -vcodec copy -acodec copy -vbsf h264_mp4toannexb 2.ts
